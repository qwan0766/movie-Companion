## 四、测试实现

### 4.1 E2E 测试设计

```python
class TestE2EFullPath:
    """完整用户路径测试"""

    def test_find_movie_full_flow(self):
        """找片全流程"""
        result = run_query("推荐科幻电影")
        assert result["user_intent"] == "find_movie"
        assert len(result["retrieved_videos"]) >= 1
        assert result["response"]
        assert not result["errors"]

    def test_knowledge_full_flow(self):
        """知识查询全流程"""
        result = run_query("介绍一下张毅")
        assert result["user_intent"] == "ask_info"
        assert result["intent_confidence"] >= 0.70
        assert result["response"]
```

### 4.2 性能测试设计

class TestPerformance:

    def test_hybrid_search_speed(self, benchmark):

### 4.3 边界条件测试

```python
class TestEdgeCases:
    """边界情况"""

    def test_empty_string(self):
        result = run_query("   ")
        assert result["user_intent"] == "unknown"

    def test_special_chars(self):
        result = run_query("<script>alert('xss')</script>")
        assert result["response"]  # 不应崩溃

    def test_long_input(self):
        text = "推荐" + "电影" * 500
        result = run_query(text)
        assert result["response"]  # 不应崩溃
```
